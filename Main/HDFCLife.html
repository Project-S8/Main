import pandas as pd import warnings import datetime as dt pyscript.write('today', dt.date.today().strftime('%A %B %d, %Y')) warnings.filterwarnings("ignore") from pyodide.http import open_url url_content=open_url("https://raw.githubusercontent.com/aakhilanvarsadat/stocks/aakhil/Quote-Equity-HDFCLIFE-EQ-29-09-2022-to-29-03-2023%20(1).csv") stock_data=pd.read_csv(url_content) y = stock_data.close y = y.iloc[::-1] stock_features = ['OPEN '] X = stock_data[stock_features] X = X.iloc[::-1] from sklearn.ensemble import GradientBoostingRegressor stock_model = GradientBoostingRegressor() stock_model.fit(X, y) t1="TATA MOTORS" string = t1.strip() closing = stock_data['close'] last=[[closing[0]]] predictions = stock_model.predict(last) t2="Previous day's Closing Value" pyscript.write("text2",t2) t3=last t3=t3[0] t3=t3[0] pyscript.write("text3",t3) t6="Today's Predicted closing value" t7=predictions t7=int(t7) pyscript.write("text6",t6) pyscript.write("text7",t7)
